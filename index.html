شروع بازی! 
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.13;

import "@openzeppelin/contracts/token/ERC20/IERC20.sol";
import "@openzeppelin/contracts/token/ERC20/ERC20.sol";
import "@openzeppelin/contracts/utils/math/SafeMath.sol";

contract Coin is ERC20 {
    using SafeMath for uint256;
    uint256 MaxSupply = 1000000 ether; // max token supply  1 000 000

    constructor() ERC20("Abolfazl Coin", "ABOL") {
        _mint(msg.sender, MaxSupply); //send 1 000 000 tokens to the contract deployer
    }

}

اینجا کجاست؟

بازی سالیدیتی یک بازی هست که شما به عنوان یک هکر یک قرارداد هوشمند اتریوم رو هک میکنید تا درباره ی امنیت قراردادها همه چیز رو یاد بگیرید. این بازی قراره هم آموزنده باشه هم باحال! این بازی از smarx ایده برداری شده و توسط ما یعنی Abol و Erphha برای شما به زبان فارسی آماده شده. امیدواریم لذت ببرید!

چجوری بازی کنم؟

این بازی از یک سری چالش ها در دسته بندی های مختلف تشکیل شده. با گذروندن هر چالش شما امتیاز دریافت میکنید. هرچی چالش سخت تر، امتیاز بیشتر! هر چالش در قالب یک قرارداد هوشمند هست که یک فانکشن isComplete یا یک متغیر public داره. هدف ما این هست که همیشه فانکشن isComplete برای ما true برگردونه که یعنی شما چالش رو با موفقیت پشت سر گذاشتید! اگه با این چیزا حال میکنی ما یه لیدربرد هم داریم.
